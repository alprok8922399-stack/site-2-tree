/* ========================================================================== 📊 СКРИПТ УПРАВЛЕНИЯ ТАБЛИЦЕЙ РЕФЕРАЛОВ ========================================================================== */ const TABLE_API_URL = '/api'; // Глобальные переменные для состояния let expandedUsers = new Set(); let currentRefSearch = ''; // Функция обновления (выносим в window, чтобы была доступна) window.updateReferralTable = async function() { const tableBody = document.getElementById('interactiveRefTableBody'); if (!tableBody) return; try { const res = await fetch(`${TABLE_API_URL}/referrals`); const data = await res.json(); if (!data.success) return; tableBody.innerHTML = ""; // Заголовки const headerRow = document.createElement('tr'); headerRow.innerHTML = `ЛогинРефыДействие`; tableBody.appendChild(headerRow); renderUserRow(data.tree, 0, tableBody); } catch (err) { console.error("Ошибка таблицы:", err); } }; function renderUserRow(node, depth, container) { if (!node || !node.username) return; const matchesSearch = currentRefSearch === '' || node.username.toLowerCase().includes(currentRefSearch); if (matchesSearch || hasMatchingChild(node, currentRefSearch)) { const tr = document.createElement('tr'); const hasChildren = node.referrals && node.referrals.length > 0; const isExpanded = expandedUsers.has(node.username); const directRefsCount = hasChildren ? node.referrals.length : 0; const isLeader = directRefsCount >= 10; tr.innerHTML = ` ${hasChildren ? (isExpanded ? '▼ ' : '▶ ') : '• '} ${node.username} ${isLeader ? '★' : ''} ${directRefsCount} ${hasChildren ? `` : '—'} `; tr.onclick = () => { document.getElementById('tableOverlay').classList.remove('show'); if (window.findUserAndFocus) window.findUserAndFocus(node.username); }; container.appendChild(tr); if (hasChildren && (isExpanded || currentRefSearch !== '')) { node.referrals.forEach(child => renderUserRow(child, depth + 1, container)); } } } function hasMatchingChild(node, searchStr) { if (!node.referrals) return false; return node.referrals.some(c => c.username.toLowerCase().includes(searchStr) || hasMatchingChild(c, searchStr)); } window.toggleUserRow = function(username) { if (expandedUsers.has(username)) expandedUsers.delete(username); else expandedUsers.add(username); window.updateReferralTable(); }; // Привязка событий сразу после загрузки файла document.addEventListener('DOMContentLoaded', () => { document.getElementById('openTableBtn').addEventListener('click', () => { document.getElementById('tableOverlay').classList.add('show'); window.updateReferralTable(); }); document.getElementById('closeOverlayBtn').addEventListener('click', () => { document.getElementById('tableOverlay').classList.remove('show'); }); document.getElementById('refSearchBtn').addEventListener('click', () => { currentRefSearch = document.getElementById('refSearchInput').value.trim().toLowerCase(); window.updateReferralTable(); }); });
